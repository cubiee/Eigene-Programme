char makefile[] = {
	"PROGRAM = %s\n"
	"OBJECTS = %s\n"
	"#------------------------------------------------\n"
	"DEPENDFILE = .depend\n"
	"CC=gcc\n"
	"CFLAGS=-Wall -pedantic \n"
	"\n"
	".PHONY: all\n"
	"all: dep $(PROGRAM)\n"
	"\n"
	"$(PROGRAM) : $(OBJECTS)\n"
	"\t$(CC) $(OBJECTS) -o $(PROGRAM)\n"
	"\n"
	"%%.o : %%.c\n"
	"\t$(CC) -c $(CFLAGS) $< -o $@\n"
	"\n"
	".PHONY: dep\n"
	"dep: *.c\n"
	"\tgcc -MM *.c > $(DEPENDFILE)\n"
	"\n"
	"-include $(DEPENDFILE)\n"
	"\n"
	".PHONY: clean\n"
	"clean:\n"
	"\trm  -rf *.o"
};
